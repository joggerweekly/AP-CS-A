Unit 5: Writing Classes

static method
example generated by google AI with "write a java person class with  static methods"
public class Person {
    private String name;
    private int age;
    private static int populationCount = 0;

    public Person(String name, int age) {
        this.name = name;
        this.age = age;
        Person.populationCount++;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public static int getPopulationCount() {
        return Person.populationCount;
    }

    public static Person createAdult(String name) {
        return new Person(name, 18);
    }

    public static boolean isAdult(int age) {
        return age >= 18;
    }

    public static void main(String[] args) {
        Person person1 = new Person("Alice", 25);
        Person person2 = new Person("Bob", 30);

        System.out.println(person1.getName() + " is an adult: " + Person.isAdult(person1.getAge()));
        System.out.println(person2.getName() + " is an adult: " + Person.isAdult(person2.getAge()));

        Person person3 = Person.createAdult("Charlie");
         System.out.println(person3.getName() + " is an adult: " + Person.isAdult(person3.getAge()));

        System.out.println("Total population: " + Person.getPopulationCount());
    }
}
